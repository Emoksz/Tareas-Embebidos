#define _XTAL_FREQ 8000000
#include <xc.h>

void main(void) {

 
    TRISBbits.TRISB0 = 0;  
    TRISCbits.TRISC0 = 1;   

   
    PORTBbits.RB0 = 0;

    while(1){

            if(PORTCbits.RC0 == 1){
            PORTBbits.RB0 = 1;
        }


        if(PORTCbits.RC0 == 0){
            PORTBbits.RB0 = 0;
        }

    }
}
